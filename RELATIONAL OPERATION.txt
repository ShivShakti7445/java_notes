//RELATIONAL OPERATION
import java.util.Scanner;
public class relational {
    public static void main ( String args[]){
        int a,b;
        System.out.print("enter the two integer");
        Scanner sc=new Scanner(System.in);
        a=sc.nextInt();
        b=sc.nextInt();
        System.out.println(a>b);
        System.out.println(a<b);
        System.out.println(a>=b);
        System.out.println(a<=b);
        System.out.println(a!=b);
        System.out.println(a==b);
      
    }
